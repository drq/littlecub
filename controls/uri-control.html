Uri control.

Native API

var uriControl1 = LittleCub("../index.html", {
    "size" : 60
}, {
    "type" : "string",
    "format" : "uri"
}, document.getElementById('uri-control-1'));

jQuery API

var uriControl1j = $('#uri-control-1-j').lc("../index.html", {
    "size" : 60
}, {
    "type" : "string",
    "format" : "uri"
});
