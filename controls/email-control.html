Email control.

Native API

var emailControl1 = LittleCub("test@nxfinc.com", {
    "size" : 60
}, {
    "type" : "string",
    "format" : "email"
}, document.getElementById('email-control-1'));

    "size" : 60
}, {
    "type" : "string",
    "format" : "email"
});
