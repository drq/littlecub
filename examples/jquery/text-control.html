Text Control Examples With jQuery

Text control with only string data parameter.

    
var textControl1 = $('#text-control-1').lc("Sample1");

Text control with data and configs parameters.

    
var textControl2 = $('#text-control-2').lc("Sample2", {
    "type" : "text",
    "size" : 60
});

Text control with data, configs and schema parameters.

    
var textControl3 = $('#text-control-3').lc("Sample3", {
    "type" : "text",
    "size" : 60
}, {
    "type" : "string",
    "minLength" : 10
});

Text control with configs and schema parameters that have various options.

    
var textControl4 = $('#text-control-4').lc("", {
    "type" : "text",
    "size" : 60,
    "label" : "Sample 4",
    "helper" : "Enter your sample text",
    "placeholder" : "Sample Text",
    "required" : true,
    "validationEvent" : "keyup",
    "name" : "sample"
}, {
    "type" : "string",
    "title" : "Sample",
    "description" : "Enter sample text",
    "minLength" : 10,
    "maxLength" : 30
}).validate();
