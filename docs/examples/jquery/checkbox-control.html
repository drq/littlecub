Examples > jQuery > Checkbox Control

Checkbox control with only boolean data parameter.

var checkboxControl1 = $('#checkbox-control-1').lc(true);

Checkbox control with data and configs parameters.

var checkboxControl2 = $('#checkbox-control-2').lc(true, {
    "type" : "checkbox"
});

Checkbox control with data, configs and schema parameters.

var checkboxControl3 = $('#checkbox-control-3').lc(null, {
    "type" : "checkbox",
    "prompt" : "Are you a student?"
}, {
    "type" : "boolean",
    "default" : true,
    "title" : "Student"
});
