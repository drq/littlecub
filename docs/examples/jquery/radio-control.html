Examples > jQuery > Radio Button Group Control

Radio control with only schema parameter.

var radioControl1 = $('#radio-control-1').lc("giant panda", null, {
    "type" : "string",
    "enum" : ["giant panda", "polar bear", "grizzly bear"]
});

Radio control with schema and configs parameters.

var radioControl2 = $('#radio-control-2').lc("giant panda", {
    "type" : "radio"
}, {
    "type" : "string",
    "enum" : ["giant panda", "polar bear", "grizzly bear", "atlas bear", "blue bear"]
});

Radio control with data, configs and schema parameters.

var radioControl3 = $('#radio-control-3').lc("giant panda", {
    "type" : "radio",
    "label" : "Pick your favorite bear:",
    "required" : true,
    "options" : ["Giant panda", "Polar bear", "Grizzly bear", "Atlas bear", "Blue bear"]
}, {
    "type" : "string",
    "enum" : ["giant panda", "polar bear", "grizzly bear", "atlas bear", "blue bear"]
});

var radioControl4 = $('#radio-control-4').lc(5, {
    "type" : "radio",
    "label" : "Pick your favorite number:",
    "required" : true,
    "options" : {
        "1" : "One",
        "2" : "Two",
        "3" : "Three",
        "4" : "Four",
        "5" : "Five"
    }
}, {
    "type" : "string",
    "enum" : [1, 2, 3, 4, 5]
});
