Examples > Barebone > Integer Control

Integer control with only integer data parameter.

var integerControl1 = LittleCub(35, null, null,document.getElementById('integer-control-1'));

Integer control with data and configs parameters.

var integerControl2 = LittleCub(35, {
    "type" : "integer",
    "label" : "Bear Age",
    "size" : 6
}, null,document.getElementById('integer-control-2'));

Integer control with data, configs and schema parameters.

var integerControl3 = LittleCub(35, {
    "type" : "integer",
    "size" : 6
}, {
    "type" : "integer",
    "minimum" : 15,
    "maximum" : 55
},document.getElementById('integer-control-3'));

Integer control with configs and schema parameters that have various options.

var integerControl4 = LittleCub("17", {
    "type" : "integer",
    "label" : "Average Age",
    "helper" : "Enter average age of a brown bear",
    "size" : 6
}, {
    "type" : "integer",
    "minimum" : 14,
    "maximum" : 55,
    "exclusiveMinimum" : true,
    "exclusiveMaximum" : true,
    "multipleOf" : 2
},document.getElementById('integer-control-4')).validate();
