Examples > Barebone > Schema Reference

LittleCub supports schema reference.

var data = {
    "age" : 30,
    "age2" : 15
};
var schema = {
    "title" : "Profile",
    "description" : "User profile.",
    "type" : "object",
    "properties" : {
        "age" : {
            "$ref" : "#/definitions/positiveInteger",
            "title" : "Age"
        },
        "age2" : {
            "$ref" : "#/definitions/positiveInteger",
            "title" : "Age 2"
        }
    },
    "required" : ["age"],
    "definitions" : {
        "positiveInteger" : {
            "type" : "integer",
            "minimum" : 0
        }
    }
};

var configs = {
};
var objectControl = LittleCub(data, configs, schema,document.getElementById('schema-reference-1'));
